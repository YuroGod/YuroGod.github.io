void __fastcall tea(int *v, _DWORD *k)
{
  int v0; // [sp+8h] [bp-14h]
  int v1; // [sp+Ch] [bp-10h]
  int sum; // [sp+10h] [bp-Ch]
  int i; // [sp+14h] [bp-8h]

  v0 = *v;
  v1 = v[1];
  sum = 0;
  for ( i = 0; i < rounds; ++i )
  {
    sum += delta;
    v0 += ((v1 >> 5) + k[1]) ^ (16 * v1 + *k) ^ (v1 + sum);
    v1 += ((v0 >> 5) + k[3]) ^ (16 * v0 + k[2]) ^ (v0 + sum);
  }
  *v = v0;
  v[1] = v1;
}

void decrypt(uint32_t *v, uint32_t *k)
{
    int v0 = v[0], v1 = v[1], i;
    uint32_t delta = 0x9E3779B9;
    uint32_t sum = delta * 32;
    uint32_t k0 = k[0], k1 = k[1], k2 = k[2], k3 = k[3];
    for (i = 0; i < 32; i++)
    {
        v1 -= ((v0 << 4) + k2) ^ (v0 + sum) ^ ((v0 >> 5) + k3);
        v0 -= ((v1 << 4) + k0) ^ (v1 + sum) ^ ((v1 >> 5) + k1);
        sum -= delta;
    }
    v[0] = v0; v[1] = v1;
}

// xxtea
void xxtea(unsigned int* v, int n, unsigned int* key) {
    // n: array v size
    unsigned int delta = 0x9e3779b9;
    unsigned int y, z, sum;
    unsigned int rounds, e;
    if (n > 1) { // encrypt
        rounds = 6 + 52/n;
        sum = 0;
        z = v[n - 1];
        do {
            sum += delta;
            e = (sum >> 2) & 3;
            for (int p = 0; p < n - 1; p++) {
                y = v[p + 1];
                v[p] += (((z >> 5 ^ y << 2) + (y >> 3 ^ z << 4)) ^ ((sum ^ y) + (key[(p & 3) ^ e] ^ z)));
                z = v[p];
            }
            y = v[0];
            v[n - 1] += (((z >> 5 ^ y << 2) + (y >> 3 ^ z << 4)) ^ ((sum ^ y) + (key[((n - 1) & 3) ^ e] ^ z)));
            z = v[n - 1];
        } while (--rounds);
    } else if (n < -1) { //decrypt
        n = -n;
        rounds = 6 + 52/n;
        sum = rounds * delta;
        y = v[0];
        do {
            e = (sum >> 2) & 3;
            for (int p = n - 1; p > 0; p--) {
                z = v[p - 1];
                v[p] -= (((z >> 5 ^ y << 2) + (y >> 3 ^ z << 4)) ^ ((sum ^ y) + (key[(p & 3) ^ e] ^ z)));
                y = v[p];
            }
            z = v[n - 1];
            v[0] -= (((z >> 5 ^ y << 2) + (y >> 3 ^ z << 4)) ^ ((sum ^ y) + (key[e] ^ z)));
            y = v[0];
            sum -= delta;
        } while (--rounds);
    }
}
